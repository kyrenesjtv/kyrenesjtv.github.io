/**
     *  选择排序
     * @param a 数组
     * @param n 数组长度
     */
    public static void selectionSort(int[] a , int n){
        if(n < 0 ){
            return;
        }
        for(int i = 0 ; i < n-1 ; i++){
            int k = i;

            for(int j = k +1 ;j<n;j++){
                //找到最小位置的索引
                if(a[j]<a[k]){
                    k=j;
                }
            }
            //数据交换
            if(i != k ){
                int temp = a[i];
                a[i]= a[k];
                a[k]=temp;
            }
        }
    }
