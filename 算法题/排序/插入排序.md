/**
     *  插入排序
     * @param a 数组
     * @param n 数组长度
     */
    public static void insertionSort(int[] a , int n )  {
        if(n <= 0 ){
            return;
        }
        for(int i = 1 ; i < n ; i++){
            int value = a[i];
            int j=i-1;
            for(; j>=0;j--){
                //进行数据交换
                if(a[j] > value){
                    a[j+1] = a[j];
                }else {
                    break;
                }
            }
            //插入数据
            a[j+1]=value;
        }
    }
